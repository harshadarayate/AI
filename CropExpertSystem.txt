    property(soil, rich_loam_or_clay).

recommend(grapes) :-
    property(climate, hot_and_dry_summers),
    property(sunlight, very_high),
    property(soil, well_drained_loam).

    property(climate, warm),

    property(soil, loam).

    property(rainfall, medium),
    property(soil, sandy_loam).
